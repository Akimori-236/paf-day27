mongo syntax

```
db.friends.insert({
    name: "fred",
    age: 18
})
db.friends.find({name: "fred"})

db.friends.updateOne({
    name: {$regex: "red", $options: "i"}
},{
    $set: {age: 21}  
})

db.friends.updateOne({
    name: {$regex: "red", $options: "i"}
},{
    $inc: {age: 1}  
})

db.friends.updateOne({
    name: {$regex: "red", $options: "i"}
},{
    $push: {hobbies: "photography"}  
})

db.friends.updateOne({
    name: {$regex: "red", $options: "i"}
},{
    $pop: {hobbies: -1}  
})
```
